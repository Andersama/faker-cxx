24template <class D>
25concept Distribution = requires {
26 // check that the result of the distribution is the same as the result of the operator()
27 // also checks for existence of the result_type and param_type nested types, as well as the operator()
28 requires std::is_same_v<typename D::result_type,
29 decltype(std::declval<D>()(
30 std::declval<typename D::param_type&>()))>;
31
32 // check that the distribution has a min(), max() and reset() member functions
33 {
34 std::declval<D>().min()
35 } -> std::same_as<typename D::result_type>;
36 {
37 std::declval<D>().max()
38 } -> std::same_as<typename D::result_type>;
39 {
40 std::declval<D>().reset()
41 } -> std::same_as<void>;
42
43 // check that the distribution is equality comparable, copy constructible and copy assignable
44 {
45 std::equality_comparable<D>
46 };
47 {
48 std::is_copy_constructible_v<D>
49 };
50 {
51 std::is_copy_assignable_v<D>
52 };
53 // check that the parameter type is equality comparable, copy constructible and copy assignable
54 {
55 std::equality_comparable<typename D::param_type>
56 };
57 {
58 std::is_copy_constructible_v<typename D::param_type>
59 };
60 {
61 std::is_copy_assignable_v<typename D::param_type>
62 };
63
64 // Check that the distribution has a param() -> param_type
65 // and param(const param_type&) member functions
66 {
67 std::declval<D>().param()
68 } -> std::same_as<typename D::param_type>;
69 {
70 std::declval<D>().param(std::declval<typename D::param_type&>())
71 } -> std::same_as<void>;
72
73 // check that D is printable to std::ostream
74 {
75 std::declval<std::ostream&>() << std::declval<D>()
76 } -> std::same_as<std::ostream&>;
77
78 // check that D can receive a value from std::istream
79 {
80 std::declval<std::istream&>() >> std::declval<D&>()
81 } -> std::same_as<std::istream&>;
82 };
