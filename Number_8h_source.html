28concept Distribution = requires {
29 // check that the result of the distribution is the same as the result of the operator()
30 // also checks for existence of the result_type and param_type nested types, as well as the operator()
31 requires std::is_same_v<typename D::result_type,
32 decltype(std::declval<D>()(
33 std::declval<typename D::param_type&>()))>;
34
35 // check that the distribution has a min(), max() and reset() member functions
36 {
37 std::declval<D>().min()
38 } -> std::same_as<typename D::result_type>;
39 {
40 std::declval<D>().max()
41 } -> std::same_as<typename D::result_type>;
42 {
43 std::declval<D>().reset()
44 } -> std::same_as<void>;
45
46 // check that the distribution is equality comparable, copy constructible and copy assignable
47 {
48 std::equality_comparable<D>
49 };
50 {
51 std::is_copy_constructible_v<D>
52 };
53 {
54 std::is_copy_assignable_v<D>
55 };
56 // check that the parameter type is equality comparable, copy constructible and copy assignable
57 {
58 std::equality_comparable<typename D::param_type>
59 };
60 {
61 std::is_copy_constructible_v<typename D::param_type>
62 };
63 {
64 std::is_copy_assignable_v<typename D::param_type>
65 };
66
67 // Check that the distribution has a param() -> param_type
68 // and param(const param_type&) member functions
69 {
70 std::declval<D>().param()
71 } -> std::same_as<typename D::param_type>;
72 {

74 } -> std::same_as<void>;

84 } -> std::same_as<std::istream&>;
